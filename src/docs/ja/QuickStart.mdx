# クイックスタートガイド

reeljsを使って最小構成のスロットゲームを5ステップで構築します。

## アーキテクチャ概要

## ステップ1: シンボル & ReelStrip 定義

各リールのシンボルと配列を定義します。

```ts
import type { ReelConfig } from 'reeljs';

type Sym = 'cherry' | 'bell' | 'bar' | 'seven' | 'replay';

const reelConfigs: ReelConfig<Sym>[] = [
  {
    symbols: [
      { id: 'cherry', name: 'チェリー', weight: 10 },
      { id: 'bell', name: 'ベル', weight: 8 },
      { id: 'bar', name: 'BAR', weight: 4 },
      { id: 'seven', name: '7', weight: 2 },
      { id: 'replay', name: 'リプレイ', weight: 6 },
    ],
    reelStrip: ['cherry', 'bell', 'bar', 'seven', 'replay', 'bell', 'cherry'],
  },
  // ... リール2、リール3も同様に定義
];
```

## ステップ2: PayTable & Payline 設定

配当の組み合わせと判定ラインを定義します。

```ts
import type { PayTable, Payline } from 'reeljs';

const payTable: PayTable<Sym> = {
  entries: [
    { pattern: ['cherry', 'cherry', 'cherry'], payout: 10, roleType: 'SMALL_WIN' },
    { pattern: ['bell', 'bell', 'bell'], payout: 8, roleType: 'SMALL_WIN' },
    { pattern: ['bar', 'bar', 'bar'], payout: 15, roleType: 'SMALL_WIN' },
    { pattern: ['seven', 'seven', 'seven'], payout: 0, roleType: 'BONUS' },
    { pattern: ['replay', 'replay', 'replay'], payout: 0, roleType: 'REPLAY' },
  ],
};

const paylines: Payline[] = [
  { index: 0, positions: [0, 0, 0] }, // 上段
  { index: 1, positions: [1, 1, 1] }, // 中段
  { index: 2, positions: [2, 2, 2] }, // 下段
];
```

## ステップ3: useSlotMachine セットアップ

メインフックでスピンエンジン・クレジット・ゲームサイクルを統合します。

```tsx
import { useSlotMachine } from 'reeljs';

function SlotGame() {
  const { spinState, spinResult, spin, reset } = useSlotMachine({
    spinEngine: { reelConfigs, payTable, paylines },
    credit: { initialCredit: 1000, betOptions: [1, 2, 3], defaultBet: 3 },
  });

  return (
    <div>
      <p>状態: {spinState}</p>
      {spinResult && <p>配当: {spinResult.totalPayout}</p>}
      <button onClick={spin} disabled={spinState === 'spinning'}>スピン</button>
      <button onClick={reset}>リセット</button>
    </div>
  );
}
```

## ステップ4: クレジット管理追加

`useCredit` で詳細なクレジット制御を行います。

```tsx
import { useCredit } from 'reeljs';

function CreditPanel() {
  const { balance, currentBet, canSpin, setBet, deposit, withdraw } = useCredit({
    initialCredit: 1000,
    betOptions: [1, 2, 3],
    defaultBet: 3,
  });

  return (
    <div>
      <p>残高: {balance} / BET: {currentBet}</p>
      <button onClick={() => setBet(1)}>1枚BET</button>
      <button onClick={() => setBet(3)}>MAX BET</button>
      <button onClick={() => deposit(100)}>+100枚</button>
    </div>
  );
}
```

## ステップ5: ゲームモード追加

モード遷移を統合して完全なゲームループを構築します。

```tsx
import { useGameMode } from 'reeljs';

const { currentMode, transition } = useGameMode({
  modeTransitionConfig: {
    normalToChance: 0.02,
    chanceTobt: 0.3,
    btToSuperBigBonus: 0.1,
  },
});
```

## 完全な最小構成サンプル

```tsx
import { useSlotMachine } from 'reeljs';
import type { ReelConfig, PayTable, Payline } from 'reeljs';

type S = 'cherry' | 'bell' | 'replay';

const reelConfigs: ReelConfig<S>[] = [
  { symbols: [{ id: 'cherry', name: 'チェリー', weight: 10 }, { id: 'bell', name: 'ベル', weight: 8 }, { id: 'replay', name: 'リプレイ', weight: 6 }], reelStrip: ['cherry', 'bell', 'replay', 'cherry', 'bell'] },
  { symbols: [{ id: 'cherry', name: 'チェリー', weight: 10 }, { id: 'bell', name: 'ベル', weight: 8 }, { id: 'replay', name: 'リプレイ', weight: 6 }], reelStrip: ['bell', 'cherry', 'replay', 'bell', 'cherry'] },
  { symbols: [{ id: 'cherry', name: 'チェリー', weight: 10 }, { id: 'bell', name: 'ベル', weight: 8 }, { id: 'replay', name: 'リプレイ', weight: 6 }], reelStrip: ['replay', 'bell', 'cherry', 'bell', 'cherry'] },
];

const payTable: PayTable<S> = {
  entries: [
    { pattern: ['cherry', 'cherry', 'cherry'], payout: 10, roleType: 'SMALL_WIN' },
    { pattern: ['bell', 'bell', 'bell'], payout: 8, roleType: 'SMALL_WIN' },
    { pattern: ['replay', 'replay', 'replay'], payout: 0, roleType: 'REPLAY' },
  ],
};

const paylines: Payline[] = [{ index: 0, positions: [1, 1, 1] }];

function MySlotGame() {
  const { spinState, spinResult, spin, reset } = useSlotMachine({
    spinEngine: { reelConfigs, payTable, paylines },
    credit: { initialCredit: 1000, betOptions: [3], defaultBet: 3 },
  });

  return (
    <div>
      <p>状態: {spinState} | 配当: {spinResult?.totalPayout ?? 0}</p>
      <button onClick={spin} disabled={spinState === 'spinning'}>スピン</button>
      <button onClick={reset}>リセット</button>
    </div>
  );
}
```
